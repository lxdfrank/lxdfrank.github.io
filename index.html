#-- coding: utf-8 --
import socket
import sys, getopt
def main():
address = (‘阿里云公网IP’,31508)
s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
while True:
inputDATA = input(“please input send data:”)

else:
inputDATA = str(inputDATA)
